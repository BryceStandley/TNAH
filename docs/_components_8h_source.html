746 void ProcessMouseRotation(TransformComponent& transform)
747 {
748 const auto [fst, snd] = Input::GetMousePos();
749 if (m_FirstMouseInput)
750 {
751 m_LastMouseXPos = fst;
752 m_LastMouseYPos = snd;
753 m_FirstMouseInput = false;
754 }
755 float offsetX = fst - m_LastMouseXPos;
756 float offsetY = m_LastMouseYPos - snd;
757 m_LastMouseXPos = fst;
758 m_LastMouseYPos = snd;
759 offsetX *= RotationSensitivity;
760 offsetY *= RotationSensitivity;
761 transform.Rotation.x += offsetX;
762 transform.Rotation.y -= offsetY;
763 if (transform.Rotation.y > 89.0f)
764 {
765 transform.Rotation.y = 89.0f;
766 }
767 if (transform.Rotation.y < -89.0f)
768 {
769 transform.Rotation.y = -89.0f;
770 }
771 }

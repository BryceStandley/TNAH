166 enum OPERATION
167 {
168 TRANSLATE_X = (1u << 0),
169 TRANSLATE_Y = (1u << 1),
170 TRANSLATE_Z = (1u << 2),
171 ROTATE_X = (1u << 3),
172 ROTATE_Y = (1u << 4),
173 ROTATE_Z = (1u << 5),
174 ROTATE_SCREEN = (1u << 6),
175 SCALE_X = (1u << 7),
176 SCALE_Y = (1u << 8),
177 SCALE_Z = (1u << 9),
178 BOUNDS = (1u << 10),
179 TRANSLATE = TRANSLATE_X | TRANSLATE_Y | TRANSLATE_Z,
180 ROTATE = ROTATE_X | ROTATE_Y | ROTATE_Z | ROTATE_SCREEN,
181 SCALE = SCALE_X | SCALE_Y | SCALE_Z
182 };
183
184 inline OPERATION operator|(OPERATION lhs, OPERATION rhs)
185 {
186 return static_cast<OPERATION>(static_cast<int>(lhs) | static_cast<int>(rhs));
187 }
